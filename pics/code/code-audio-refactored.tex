		interface AudioOutput {
			void play(String sound);
		}
		
		class AudioManager {
			List<AudioOutput> outputs = new ArrayList<>();
			
			void registerOutput(AudioOutput output) {
				outputs.add(output);
			}
			
			void playSound(String sound) {
				for (AudioOutput output : outputs) {
					output.play(sound);
				}
			}
		}
	\end{lstlisting}
