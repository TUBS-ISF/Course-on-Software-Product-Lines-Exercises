		public aspect PaymentAspect {
			pointcut paymentMethods(String user, double amount) :
			execution(* PaymentProcessor.processPayment(String, double)) &&
			args(user, amount);
			
			before(String user, double amount) : paymentMethods(user, amount) {
				System.out.println("Start processing for " + user);
			}
			
			after(String user, double amount) : paymentMethods(user, amount) {
				System.out.println("Finished processing for " + user);
			}
			
			around() : execution(* PaymentProcessor.processPayment(..)) {
				try {
					proceed();
				} catch (Exception e) {
					System.err.println("Advice caught error: " + e.getMessage());
				}
			}
		}
